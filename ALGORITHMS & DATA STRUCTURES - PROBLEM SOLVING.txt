var num = [12, 21, 43, 5, 0, 12];
function sumOfArray(arr) {
  var sum = 0;
  for (let i = 0; i < arr.length; i++) {
    sum += arr[0];
  }
  return sum;
}
console.log(sumOfArray(num));
