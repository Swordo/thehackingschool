var num = [12, 21, 43, 5, 0, 12];
function sumOfArray(arr) {
  var sum = 0;
  for (let i = 0; i < arr.length; i++) {
    sum += arr[0];
  }
  return sum;
}
console.log(sumOfArray(num));

var smax;
function maxNumber(arr) {
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] > max) {
      max = arr[i];
    }
  }
  return max;
}

function secMax(ar) {
  for (let i = 0; i < ar.length; i++) {
    if (ar[i] > max) {
      smax = max;
      max = ar[i];
    }
  }

}
