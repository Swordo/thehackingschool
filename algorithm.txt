function q5(a,b){
    var c = 0;
    for(var i = 1 ;i<a; i++){
        if(b >= a/i ){
            break;
        }
    }
    if(a/i == b){
        c = 0;
    }else{
        c = a - (i-1)*b;
        
    }
    return c;
}
        console.log(q5(17,17));

// formule for x power y (x**y)
function power(num1, num2) {
  var ans = 1;
  for (let i = 0; i < num2; i++) {
    ans *= num1;
  }
  return ans;
}
console.log(power(2, 0));
