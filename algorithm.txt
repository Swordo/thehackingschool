function q5(a,b){
    var c = 0;
    for(var i = 1 ;i<a; i++){
        if(b >= a/i ){
            break;
        }
    }
    if(a/i == b){
        c = 0;
    }else{
        c = a - (i-1)*b;
        
    }
    return c;
}

// formula for x power y (x**y)
function power(num1, num2) {
  var ans = 1;
  for (let i = 0; i < num2; i++) {
    ans *= num1;
  }
  return ans;
}
console.log(power(2, 0));

function fib(n){
    var num = [0,1];
    if(n == 0){
        return 0;
        }
    else if(n == 1){
        return fib[0];
    }
    else if ( n>=2){
        for(let i = 0 ; i <n-2 ; i++ ){
                num.push(num[i]+num[i+1]);
        }
    } 
    return num;
    
}
console.log(fib(7));
